---
title: "Consultar Nota Fiscal"
api: "GET /invoices/{id}"
description: "Retorna os detalhes de uma nota fiscal específica"
---

## Request

<ParamField path="id" type="string" required>
  ID da nota fiscal.
</ParamField>

## Response

<ResponseField name="id" type="string">
  ID único da nota.
</ResponseField>

<ResponseField name="company_id" type="string">
  ID da empresa emissora.
</ResponseField>

<ResponseField name="status" type="string">
  Status da nota: `processing`, `authorized`, `rejected`, `cancelled`, `error`.
</ResponseField>

<ResponseField name="numero_nota" type="string">
  Número da nota fiscal (quando autorizada).
</ResponseField>

<ResponseField name="valor" type="number">
  Valor da nota.
</ResponseField>

<ResponseField name="descricao" type="string">
  Descrição do serviço.
</ResponseField>

<ResponseField name="external_id" type="string">
  ID externo informado na emissão.
</ResponseField>

<ResponseField name="tomador" type="object">
  Dados do tomador.
</ResponseField>

<ResponseField name="pdf_url" type="string">
  URL para download do PDF (quando autorizada).
</ResponseField>

<ResponseField name="xml_url" type="string">
  URL para download do XML (quando autorizada).
</ResponseField>

<ResponseField name="erro_mensagem" type="string">
  Mensagem de erro (quando rejeitada ou com erro).
</ResponseField>

<ResponseField name="created_at" type="string">
  Data de criação.
</ResponseField>

<ResponseField name="authorized_at" type="string">
  Data de autorização.
</ResponseField>

```bash
  -H "X-API-Key: ez_live_abc123_sk_7f8a9b2c4d6e1f3a5b7c9d0e2f4a6b8c"
```
</RequestExample>
