---
title: "Listar Empresas"
api: "GET /companies"
description: "Lista todas as empresas cadastradas pelo parceiro"
---

## Request

<ParamField query="page" type="integer">
  Número da página (padrão: 1).
</ParamField>

<ParamField query="per_page" type="integer">
  Itens por página (padrão: 15, máximo: 100).
</ParamField>

## Response

<ResponseField name="data" type="array">
  Lista de empresas.

  <Expandable title="Campos da empresa">
    <ResponseField name="id" type="string">
      ID único da empresa.
    </ResponseField>
    <ResponseField name="cnpj" type="string">
      CNPJ da empresa.
    </ResponseField>
    <ResponseField name="razao_social" type="string">
      Razão social da empresa.
    </ResponseField>
    <ResponseField name="nome_fantasia" type="string">
      Nome fantasia da empresa.
    </ResponseField>
    <ResponseField name="municipio" type="string">
      Município da empresa.
    </ResponseField>
    <ResponseField name="uf" type="string">
      UF da empresa.
    </ResponseField>
    <ResponseField name="regime_tributario" type="string">
      Regime tributário.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Data de criação.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Informações de paginação.

  <Expandable title="Campos de meta">
    <ResponseField name="current_page" type="integer">
      Página atual.
    </ResponseField>
    <ResponseField name="last_page" type="integer">
      Última página.
    </ResponseField>
    <ResponseField name="per_page" type="integer">
      Itens por página.
    </ResponseField>
    <ResponseField name="total" type="integer">
      Total de registros.
    </ResponseField>
  </Expandable>
</ResponseField>
