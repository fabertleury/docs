---
title: "Criar Empresa"
api: "POST /companies"
description: "Cadastra uma nova empresa para emissão de notas fiscais"
---

## Request

<ParamField body="cnpj" type="string" required>
  CNPJ da empresa (com ou sem formatação).
</ParamField>

<ParamField body="razao_social" type="string" required>
  Razão social da empresa.
</ParamField>

<ParamField body="regime_tributario" type="string" required>
  Regime tributário da empresa. Valores aceitos: `mei`, `simples_nacional`, `lucro_presumido`, `lucro_real`.
</ParamField>

<ParamField body="municipio" type="string" required>
  Município da empresa.
</ParamField>

<ParamField body="uf" type="string" required>
  UF (estado) da empresa. Ex: `SP`, `RJ`, `MG`.
</ParamField>

<ParamField body="nome_fantasia" type="string">
  Nome fantasia da empresa.
</ParamField>

<ParamField body="email" type="string">
  E-mail da empresa.
</ParamField>

<ParamField body="telefone" type="string">
  Telefone da empresa.
</ParamField>

<ParamField body="logradouro" type="string">
  Endereço (rua/avenida) da empresa.
</ParamField>

<ParamField body="numero" type="string">
  Número do endereço.
</ParamField>

<ParamField body="inscricao_municipal" type="string">
  Inscrição municipal da empresa.
</ParamField>

<ParamField body="inscricao_estadual" type="string">
  Inscrição estadual da empresa.
</ParamField>

## Response

<ResponseField name="id" type="string">
  ID único da empresa criada.
</ResponseField>

<ResponseField name="cnpj" type="string">
  CNPJ da empresa.
</ResponseField>

<ResponseField name="razao_social" type="string">
  Razão social da empresa.
</ResponseField>

<ResponseField name="nome_fantasia" type="string">
  Nome fantasia da empresa.
</ResponseField>

<ResponseField name="regime_tributario" type="string">
  Regime tributário.
</ResponseField>

<ResponseField name="municipio" type="string">
  Município.
</ResponseField>

<ResponseField name="uf" type="string">
  UF.
</ResponseField>

<ResponseField name="created_at" type="string">
  Data de criação.
</ResponseField>
