---
title: "Listar Notas Fiscais"
api: "GET /invoices"
description: "Lista as notas fiscais emitidas com filtros opcionais"
---

## Request

<ParamField query="company_id" type="string">
  Filtrar por empresa.
</ParamField>

<ParamField query="status" type="string">
  Filtrar por status. Valores: `processing`, `authorized`, `rejected`, `cancelled`, `error`.
</ParamField>

<ParamField query="from" type="string">
  Data inicial no formato `YYYY-MM-DD`.
</ParamField>

<ParamField query="to" type="string">
  Data final no formato `YYYY-MM-DD`.
</ParamField>

<ParamField query="page" type="integer">
  Número da página (padrão: 1).
</ParamField>

<ParamField query="per_page" type="integer">
  Itens por página (padrão: 15, máximo: 100).
</ParamField>

## Response

<ResponseField name="data" type="array">
  Lista de notas fiscais.

  <Expandable title="Campos da nota fiscal">
    <ResponseField name="id" type="string">
      ID único da nota.
    </ResponseField>
    <ResponseField name="company_id" type="string">
      ID da empresa emissora.
    </ResponseField>
    <ResponseField name="status" type="string">
      Status da nota.
    </ResponseField>
    <ResponseField name="numero_nota" type="string">
      Número da nota fiscal (quando autorizada).
    </ResponseField>
    <ResponseField name="valor" type="number">
      Valor da nota.
    </ResponseField>
    <ResponseField name="external_id" type="string">
      ID externo informado na emissão.
    </ResponseField>
    <ResponseField name="tomador" type="object">
      Dados do tomador.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Data de criação.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Informações de paginação.
</ResponseField>
