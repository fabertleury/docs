---
title: "Listar Tomadores"
api: "GET /companies/{id}/customers"
description: "Lista os tomadores (clientes) cadastrados de uma empresa"
---

## Request

<ParamField path="id" type="string" required>
  ID da empresa.
</ParamField>

<ParamField query="page" type="integer">
  Número da página (padrão: 1).
</ParamField>

<ParamField query="per_page" type="integer">
  Itens por página (padrão: 15, máximo: 100).
</ParamField>

## Response

<ResponseField name="data" type="array">
  Lista de tomadores.

  <Expandable title="Campos do tomador">
    <ResponseField name="id" type="string">
      ID único do tomador.
    </ResponseField>
    <ResponseField name="cpf_cnpj" type="string">
      CPF ou CNPJ do tomador.
    </ResponseField>
    <ResponseField name="razao_social" type="string">
      Nome ou razão social do tomador.
    </ResponseField>
    <ResponseField name="email" type="string">
      E-mail do tomador.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Data de criação.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Informações de paginação.

  <Expandable title="Campos de meta">
    <ResponseField name="current_page" type="integer">
      Página atual.
    </ResponseField>
    <ResponseField name="last_page" type="integer">
      Última página.
    </ResponseField>
    <ResponseField name="per_page" type="integer">
      Itens por página.
    </ResponseField>
    <ResponseField name="total" type="integer">
      Total de registros.
    </ResponseField>
  </Expandable>
</ResponseField>

